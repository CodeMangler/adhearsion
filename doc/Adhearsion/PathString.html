def from_application_subdirectory(folder)
  folder = File.expand_path folder
  ahn_rc = nil

  until ahn_rc || folder == "/"
    possible_ahn_rc = File.join(folder, ".ahnrc")
    if File.exists?(possible_ahn_rc)
      ahn_rc = possible_ahn_rc
    else
      folder = File.expand_path(folder + "/..")
    end
  end
  ahn_rc ? new(folder) : nil
end
