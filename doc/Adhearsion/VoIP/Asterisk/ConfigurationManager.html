def normalize_configuration(file_contents)
  # cat sip.conf | sed -e 's/\s*;.*$//g' | sed -e '/^;.*$/d' | sed -e '/^\s*$/d'
  file_contents.split(/\n+/).map do |line|
    line.sub(/;.+$/, '').strip
  end.join("\n").squeeze("\n")
end
