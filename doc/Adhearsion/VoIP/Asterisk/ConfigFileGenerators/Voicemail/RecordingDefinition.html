def allowed_length(seconds)
  case seconds
    when Fixnum
      int :maxmessage => "value"
    when Range
      int :minmessage => seconds.first
      int :maxmessage => seconds.last
    else
      raise ArgumentError, "Argument must be a Fixnum or Range!"
  end
end
