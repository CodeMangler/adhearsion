def _(pattern)
  # Uncomment the following code fragment for complete compatibility.
  # The fragment handles the seldom-used hyphen number spacer with no
  # meaning.
	Regexp.new '^' << pattern.# gsub(/(?!\[[\w+-]+)-(?![\w-]+\])/,'').
	  gsub('X', '[0-9]').gsub('Z', '[1-9]').gsub('N','[2-9]').
	  gsub('.','.+').gsub('!','.*') << '$'
end
