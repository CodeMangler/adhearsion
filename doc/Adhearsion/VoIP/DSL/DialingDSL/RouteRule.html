def >>(other)
  case other
    when RouteRule then merge! other
    when ProviderDefinition then providers << other
    else raise RouteException, "Unrecognized object in route definition: #{other.inspect}"
  end
  self
end

def |(other)

    when RouteRule then merge! other
    when Regexp
      patterns << other
      self
    else raise other.inspect
  end
