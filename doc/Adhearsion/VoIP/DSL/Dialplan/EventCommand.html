def initialize(app, *args, &block)
  @hash = args.pop if args.last.kind_of?(Hash)
  @app, @args = app, args

  if @hash
    @returns = @hash[:returns] || :raw
    @on_keypress = @hash[:on_keypress]
  end

  @response_block = block if block_given?
end
